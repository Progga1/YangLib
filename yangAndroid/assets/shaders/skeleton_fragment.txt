#ANDROID precision mediump float;
uniform sampler2D texSampler;

varying vec2 texCoord;
varying vec4 color;
varying vec4 addColor;

void main()
{
	#NOPREMULT gl_FragColor = texture2D(texSampler, vec2(texCoord.x,texCoord.y)) * color + addColor;
	#PREMULT vec4 texCl = texture2D(texSampler, vec2(texCoord.x,texCoord.y));
	#PREMULT gl_FragColor = texCl * color + addColor*texCl.a;
}
